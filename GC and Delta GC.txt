#!/usr/bin/env python3
"""
GC content and compositional (delta-GC and GC-skew) analysis.

Features:
- Accepts one or more FASTA or GenBank (.gb/.gbk/.genbank) files.
- Computes:
  - Global GC fraction per sequence / record.
  - Windowed GC fraction.
  - Delta-GC = windowed_GC - global_GC.
  - GC-skew = (G - C) / (G + C) per window.
- Exports:
  - Summary table by record (CSV).
  - Window-level signals table (CSV).
- Optional plots of delta-GC and GC-skew tracks as PNGs.

Dependencies:
  pip install biopython matplotlib pandas numpy

Example usage:
  python gc_delta_analysis.py \\
      --input ASM1993100_hallmarkR1.fasta ASM1993100_hallmarkR1.gbk \\
      --window 1000 --step 100 \\
      --out-prefix gc_analysis \\
      --plot

"""

import argparse
import os
import sys

import numpy as np
import pandas as pd
from Bio import SeqIO
import matplotlib.pyplot as plt


def seq_to_str(seq_obj):
    """Convert a Biopython Seq object to an uppercase string."""
    return str(seq_obj).upper()


def gc_fraction(seq_str):
    """Return GC fraction = (G + C) / (A + T + G + C) for a sequence string."""
    s = seq_str.upper()
    g = s.count("G")
    c = s.count("C")
    a = s.count("A")
    t = s.count("T")
    denom = a + t + g + c
    if denom == 0:
        return np.
    return (g + c) / float(denom)


def gc_skew_value(seq_str):
    """Return GC skew = (G - C) / (G + C) for a sequence string."""
    s = seq_str.upper()
    g = s.count("G")
    c = s.count("C")
    denom = g + c
    if denom == 0:
        return np.
    return (g - c) / float(denom)


def sliding_windows(seq_str, window, step):
    """
    Generate sliding windows over a sequence string.

    Returns:
      starts: list of window start positions (0-based).
      centers: list of window centers (float, 1-based-ish).
      seqs: list of window subsequences.
    """
    n = len(seq_str)
    if n == 0 or window <= 0 or step <= 0:
        return [], [], []

    starts = list(range(0, max(n - window + 1, 0), step))
    centers = [s + window / 2.0 for s in starts]
    seqs = [seq_str[s:s + window] for s in starts]
    return starts, centers, seqs


def compute_signals_for_sequence(seq_str, window, step):
    """
    Compute global GC and windowed GC, delta-GC, GC-skew for one sequence.

    Returns:
      global_gc (float)
      signals_df (DataFrame) with columns:
        window_start, window_center, gc, delta_gc, gc_skew
    """
    starts, centers, win_seqs = sliding_windows(seq_str, window, step)
    if len(win_seqs) == 0:
        global_gc = gc_fraction(seq_str)
        empty_df = pd.DataFrame(
            {
                "window_start": [],
                "window_center": [],
                "gc": [],
                "delta_gc": [],
                "gc_skew": [],
            }
        )
        return global_gc, empty_df

    gc_vals = np.array([gc_fraction(w) for w in win_seqs], dtype=float)
    skew_vals = np.array([gc_skew_value(w) for w in win_seqs], dtype=float)

    global_gc = gc_fraction(seq_str)
    delta_gc = gc_vals - global_gc

    signals_df = pd.DataFrame(
        {
            "window_start": starts,
            "window_center": centers,
            "gc": gc_vals,
            "delta_gc": delta_gc,
            "gc_skew": skew_vals,
        }
    )
    return global_gc, signals_df


def detect_format_from_extension(path):
    """Guess sequence file format from extension."""
    lower = path.lower()
    if lower.endswith(".fa") or lower.endswith(".fasta") or lower.endswith(".fna"):
        return "fasta"
    if lower.endswith(".gb") or lower.endswith(".gbk") or lower.endswith(".genbank"):
        return "genbank"
    # Default fallback
    return "fasta"


def parse_input_files(file_paths):
    """
    Parse sequences from provided files using Biopython SeqIO.

    Returns:
      seq_records_df: DataFrame with columns:
        file, record_id, length_bp, sequence
    """
    rows = []
    for path in file_paths:
        if not os.path.isfile(path):
            sys.stderr.write("Warning: file not found: " + path + "\n")
            continue
        fmt = detect_format_from_extension(path)
        try:
            for rec in SeqIO.parse(path, fmt):
                seq_str = seq_to_str(rec.seq)
                rows.append(
                    {
                        "file": os.path.basename(path),
                        "record_id": rec.id,
                        "length_bp": len(seq_str),
                        "sequence": seq_str,
                    }
                )
        except Exception as e:
            sys.stderr.write(
                "Error reading " + path + " with format " + fmt + ": " + str(e) + "\n"
            )
    if not rows:
        sys.stderr.write("No sequences parsed from input files.\n")
        return pd.DataFrame(columns=["file", "record_id", "length_bp", "sequence"])
    return pd.DataFrame(rows)


def analyze_sequences(seq_df, window, step):
    """
    Run GC / delta-GC / GC-skew analysis on all sequences in seq_df.

    Returns:
      summary_df: record-level summary (with GC and delta-GC/GC-skew stats).
      signals_df: window-level signals across all records.
    """
    summary_rows = []
    signals_list = []

    for idx in range(seq_df.shape[0]):
        file_name = seq_df.loc[idx, "file"]
        rec_id = seq_df.loc[idx, "record_id"]
        seq_str = seq_df.loc[idx, "sequence"]

        global_gc, sig_df = compute_signals_for_sequence(seq_str, window, step)

        sig_df["file"] = file_name
        sig_df["record_id"] = rec_id
        sig_df["length_bp"] = len(seq_str)
        sig_df["global_gc"] = global_gc
        signals_list.append(sig_df)

        # Summary stats per record
        if sig_df.shape[0] > 0:
            delta_gc_vals = sig_df["delta_gc"].to_numpy(dtype=float)
            gc_skew_vals = sig_df["gc_skew"].to_numpy(dtype=float)
            delta_gc_mean = float(np.nanmean(delta_gc_vals))
            delta_gc_sd = float(np.nanstd(delta_gc_vals))
            delta_gc_min = float(np.nanmin(delta_gc_vals))
            delta_gc_max = float(np.nanmax(delta_gc_vals))
            gc_skew_mean = float(np.nanmean(gc_skew_vals))
            gc_skew_sd = float(np.nanstd(gc_skew_vals))
        else:
            # For very short sequences where no windows are produced
            delta_gc_mean = np.
            delta_gc_sd = np.
            delta_gc_min = np.
            delta_gc_max = np.
            gc_skew_mean = np.
            gc_skew_sd = np.

        summary_rows.append(
            {
                "file": file_name,
                "record_id": rec_id,
                "length_bp": len(seq_str),
                "global_gc": global_gc,
                "delta_gc_mean": delta_gc_mean,
                "delta_gc_sd": delta_gc_sd,
                "delta_gc_min": delta_gc_min,
                "delta_gc_max": delta_gc_max,
                "gc_skew_mean": gc_skew_mean,
                "gc_skew_sd": gc_skew_sd,
            }
        )

    summary_df = pd.DataFrame(summary_rows)
    if signals_list:
        signals_df = pd.concat(signals_list, ignore_index=True)
    else:
        signals_df = pd.DataFrame(
            columns=[
                "window_start",
                "window_center",
                "gc",
                "delta_gc",
                "gc_skew",
                "file",
                "record_id",
                "length_bp",
                "global_gc",
            ]
        )

    return summary_df, signals_df


def make_plots(signals_df, out_prefix, window, step):
    """
    Generate delta-GC and GC-skew plots per record as PNG files.

    One pair of PNGs per record_id:
      <out_prefix>_<record_id>_deltaGC.png
      <out_prefix>_<record_id>_GCskew.png
    """
    if signals_df.empty:
        sys.stderr.write("No signals available for plotting.\n")
        return

    # Ensure output directory exists
    out_dir = os.path.dirname(out_prefix)
    if out_dir and not os.path.isdir(out_dir):
        os.makedirs(out_dir, exist_ok=True)

    for rec_id in sorted(signals_df["record_id"].unique()):
        sub = signals_df[signals_df["record_id"] == rec_id].copy()
        sub = sub.sort_values("window_center")

        # Delta-GC plot
        plt.figure(figsize=(10, 4))
        plt.plot(sub["window_center"], sub["delta_gc"], linewidth=1)
        plt.axhline(0, color="black", linewidth=1)
        plt.xlabel("Position (bp, window center)")
        plt.ylabel("Delta GC")
        plt.title(
            "Delta-GC (windowed GC - global GC)\n"
            + rec_id
            + " | window="
            + str(window)
            + " bp, step="
            + str(step)
            + " bp"
        )
        plt.tight_layout()
        delta_out = out_prefix + "_" + rec_id + "_deltaGC.png"
        plt.savefig(delta_out, dpi=200)
        plt.close()

        # GC-skew plot
        plt.figure(figsize=(10, 4))
        plt.plot(sub["window_center"], sub["gc_skew"], linewidth=1)
        plt.axhline(0, color="black", linewidth=1)
        plt.xlabel("Position (bp, window center)")
        plt.ylabel("GC-skew (G-C)/(G+C)")
        plt.title(
            "GC-skew\n"
            + rec_id
            + " | window="
            + str(window)
            + " bp, step="
            + str(step)
            + " bp"
        )
        plt.tight_layout()
        skew_out = out_prefix + "_" + rec_id + "_GCskew.png"
        plt.savefig(skew_out, dpi=200)
        plt.close()


def main():
    parser = argparse.ArgumentParser(
        description="GC content and delta-GC / GC-skew analysis for FASTA/GenBank files."
    )
    parser.add_argument(
        "--input",
        "-i",
        nargs="+",
        required=True,
        help="Input sequence files (FASTA or GenBank).",
    )
    parser.add_argument(
        "--window",
        "-w",
        type=int,
        default=1000,
        help="Sliding-window size in bp (default: 1000).",
    )
    parser.add_argument(
        "--step",
        "-s",
        type=int,
        default=100,
        help="Sliding-window step in bp (default: 100).",
    )
    parser.add_argument(
        "--out-prefix",
        "-o",
        default="gc_analysis",
        help="Output prefix for result files (default: gc_analysis).",
    )
    parser.add_argument(
        "--plot",
        action="store_true",
        help="If set, generate PNG plots for delta-GC and GC-skew per record.",
    )

    args = parser.parse_args()

    seq_df = parse_input_files(args.input)
    if seq_df.empty:
        sys.stderr.write("No sequences to analyze; exiting.\n")
        sys.exit(1)

    print("Parsed sequences:")
    print(seq_df[["file", "record_id", "length_bp"]])

    summary_df, signals_df = analyze_sequences(seq_df, args.window, args.step)

    # Record-level summary (one row per file+record; users can de-duplicate on record_id if needed)
    summary_path = args.out_prefix + "_summary_by_record.csv"
    summary_df.to_csv(summary_path, index=False)

    # Window-level signals
    signals_path = args.out_prefix + "_window_signals.csv"
    signals_df.to_csv(signals_path, index=False)

    # Optional: a "unique-record" summary (deduplicate by record_id, keeping first occurrence)
    unique_summary_df = summary_df.drop_duplicates(subset=["record_id"]).copy()
    unique_summary_path = args.out_prefix + "_summary_unique_records.csv"
    unique_summary_df.to_csv(unique_summary_path, index=False)

    print("Wrote record-level summary to:", summary_path)
    print("Wrote unique-record summary to:", unique_summary_path)
    print("Wrote window-level signals to:", signals_path)

    if args.plot:
        print("Generating plots...")
        make_plots(signals_df, args.out_prefix, args.window, args.step)
        print("Plots written with prefix:", args.out_prefix)


if __name__ == "__main__":
    main()
